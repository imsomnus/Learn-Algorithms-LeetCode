题目：从扑克牌中随机抽5张牌，判断是不是一个顺子，即这5张牌是不是连续的。2～10为数字本身，A为1，J为11，Q为12，K为13，而大、小王为 0 ，可以看成任意数字。A 不能视为 14。

/**
 * @param {number[]} nums
 * @return {boolean}
 */
var isStraight = function(nums) {
    nums.sort((a,b) => {
        return a-b;
    }) 
    let max = 0;
    for (let i=0;i<4;i++){
        if(nums[i] === 0){
            continue;
        } else if (nums[i] === nums[i+1]){
            return false;
        } else {
            max += nums[i+1] - nums[i];
        }
    }  
    return max < 5;
};
